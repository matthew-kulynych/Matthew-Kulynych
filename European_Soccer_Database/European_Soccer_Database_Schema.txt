-- SQL Create Table Statements --

-- PLAYER
CREATE TABLE PLAYER (
	PLAYER_ID INT NOT NULL PRIMARY KEY,	-- Unique ID
	PLAYER_FNAME VARCHAR(30) NOT NULL,	-- Player's First Name
	PLAYER_LNAME VARCHAR(50) NOT NULL,	-- Player's Last Name
	PLAYER_ALIAS VARCHAR(30),		-- Player's Alias (e.g. Casemiro)
	PLAYER_NATION VARCHAR(30) NOT NULL,	-- Player's Nationality
	PLAYER_DOB DATE NOT NULL CHECK (PLAYER_DOB > '1950-01-01'),	-- Player's date of birth
	PLAYER_POS ENUM('GK', 'DEF', 'MID', 'FWD') NOT NULL,		-- Player's primary position
	PLAYER_HEIGHT SMALLINT NOT NULL CHECK(PLAYER_HEIGHT BETWEEN 0 AND 99),	-- Player's height
	PLAYER_WEIGHT SMALLINT NOT NULL CHECK(PLAYER_WEIGHT BETWEEN 0 AND 500),	-- Player's weight
	PLAYER_FOOT ENUM('L', 'R') NOT NULL,	-- Player's dominant foot
	PLAYER_ACTIVE BOOLEAN NOT NULL,		-- Flag for if player is active
	CONTRACT_ID INT				-- Player's contract ID
);

-- TEAM
CREATE TABLE TEAM(
	TEAM_ID INT NOT NULL PRIMARY KEY, 	-- Unique ID
	TEAM_NAME VARCHAR(40) NOT NULL,		-- Team's name
	TEAM_ABBV CHAR(3) NOT NULL,		-- Team's scoreboard abbreviation
	TEAM_COUNTRY VARCHAR(30) NOT NULL,	-- Team's country
	TEAM_CITY VARCHAR(30) NOT NULL,		-- Team's city
	STADIUM_ID INT NOT NULL,		-- Team's stadium ID
	MANAGER_ID INT				-- Team's manager ID
);

-- MANAGER
CREATE TABLE MANAGER(
	MANAGER_ID INT NOT NULL PRIMARY KEY,	-- Unique ID
	MANAGER_FNAME VARCHAR(30) NOT NULL,	-- Manager's first name
	MANAGER_LNAME VARCHAR(50) NOT NULL,	-- Manager's last name
	MANAGER_ALIAS VARCHAR(30),		-- Manager's alias
	MANAGER_DOB DATE NOT NULL CHECK(MANAGER_DOB > 1900-01-01),	-- Manager's date of birth
	MANAGER_NATION VARCHAR(30) NOT NULL,	-- Manager's nationality
	TEAM_ID INT,				-- Manager's team ID
	CONTRACT_ID INT				-- Manager's contract ID
);

-- STAFF
CREATE TABLE STAFF(
	STAFF_ID INT NOT NULL PRIMARY KEY,	-- Staff's unique ID
	STAFF_FNAME VARCHAR(30) NOT NULL,	-- Staff's first name
	STAFF_LNAME VARCHAR(50) NOT NULL,	-- Staff's last name
	STAFF_TYPE ENUM('Coach', 'Medical', 'Front Office', 'Scout') NOT NULL,	-- Staff subtype category
	TEAM_ID INT NOT NULL			-- Staff's team ID
);

-- COACH (subtype)
CREATE TABLE COACH(
	STAFF_ID INT NOT NULL PRIMARY KEY,	-- Coach's staff ID
	COACH_TITLE VARCHAR(50) NOT NULL,	-- Coach's title
	COACH_CERT VARCHAR(50) NOT NULL,	-- Coach's license or credentials
	FOREIGN KEY (STAFF_ID) REFERENCES STAFF(STAFF_ID)	-- Subtype foreign key constraint
);

-- MEDICAL (subtype)
CREATE TABLE MEDICAL(
	STAFF_ID INT NOT NULL PRIMARY KEY,	-- Medical's staff ID
	MED_TITLE VARCHAR(50) NOT NULL,		-- Medical's title
	MED_CERT VARCHAR(10) NOT NULL,		-- Medical's certification, license or degree
	MED_SPECIAL VARCHAR(30),		-- Medical's specialization
	MED_CERT_DATE DATE NOT NULL,		-- Medical's most recent certification date
	FOREIGN KEY (STAFF_ID) REFERENCES STAFF(STAFF_ID)	-- Subtype foreign key constraint
);

-- FRONT_OFFICE
CREATE TABLE FRONT_OFFICE(
	STAFF_ID INT NOT NULL PRIMARY KEY,	-- FO member's staff ID
	FO_TITLE VARCHAR(50) NOT NULL,		-- FO member's title
	FO_DEPT VARCHAR(30) NOT NULL,		-- FO member's department
	FOREIGN KEY (STAFF_ID) REFERENCES STAFF(STAFF_ID)	-- Subtype foreign key constraint
);

-- SCOUT
CREATE TABLE SCOUT(
	STAFF_ID INT NOT NULL PRIMARY KEY,	-- Scout's staff ID
	SCOUT_AREA VARCHAR(30),			-- Scout's area of scouting
	SCOUT_ACTIVE BOOLEAN NOT NULL,		-- Flag if scout is currently deployed
	FOREIGN KEY (STAFF_ID) REFERENCES STAFF(STAFF_ID)	-- Subtype foreign key constraint
);

-- MATCHES
CREATE TABLE MATCHES(
	MATCH_ID INT NOT NULL PRIMARY KEY,	-- Match unique ID
	MATCH_DATE DATE NOT NULL,		-- Match date
	MATCH_TIME TIME NOT NULL,		-- Match kickoff time
	MATCH_ATTENDANCE INT CHECK(MATCH_ATTENDANCE >= 0),	-- Match fan attendance
	MATCH_PLAYED BOOLEAN NOT NULL,		-- Flag for if match has occurred
	COMPETITION_ID INT NOT NULL,		-- Match's competition ID
	SEASON_ID INT NOT NULL,			-- Match's season ID
	STADIUM_ID INT NOT NULL			-- Match's stadium ID
);	

-- STADIUM
CREATE TABLE STADIUM(
	STADIUM_ID INT NOT NULL PRIMARY KEY,	-- Stadium's unique ID
	STADIUM_NAME VARCHAR(50) NOT NULL,	-- Stadium's name
	STADIUM_CITY VARCHAR(30) NOT NULL,	-- Stadium's city
	STADIUM_COUNTRY VARCHAR(30) NOT NULL,	-- Stadium's country
	STADIUM_LONG DECIMAL(9, 4) NOT NULL,	-- Stadium's longitude
	STADIUM_LAT DECIMAL(9, 4) NOT NULL,	-- Stadium's latitude
	STADIUM_CAPACITY INT NOT NULL CHECK(STADIUM_CAPACITY >= 0)	-- Stadium's fan capacity
);

-- MATCH_EVENT
CREATE TABLE MATCH_EVENT(
	EVENT_ID INT NOT NULL PRIMARY KEY,	-- Match event's unique ID
	EVENT_TYPE VARCHAR(20) NOT NULL,	-- Match event type (Goal, Foul, etc.)
	EVENT_MINUTE SMALLINT NOT NULL CHECK(EVENT_MINUTE BETWEEN 1 AND 120),	-- Minute the event occurred
	EVENT_IN_PKS BOOLEAN NOT NULL,		-- Flag for if the event was in a PK Shootout
	EVENT_DESC TEXT,			-- Text description of the event
	MATCH_ID INT NOT NULL,			-- Event's match ID
	PLAYER_ID INT NOT NULL,			-- Event's player ID
	TEAM_ID INT NOT NULL			-- Event's team ID
);

-- TRANSFER
CREATE TABLE TRANSFER(
	TRAN_ID INT NOT NULL PRIMARY KEY,	-- Transfer's unique ID
	TRAN_DATE DATE NOT NULL,		-- Transfer's date
	TRAN_AMOUNT INT NOT NULL CHECK(TRAN_AMOUNT >= 0),	-- Transfer's dollar amount
	TRAN_ADDON INT CHECK(TRAN_ADDON >= 0),	-- Total dollar amount of any add-on's/clauses
	TRAN_TYPE VARCHAR(50) NOT NULL,		-- Type of transfer
	TRAN_ISLOAN BOOLEAN NOT NULL,		-- Flag for if the transfer is a loan
	PLAYER_ID INT NOT NULL,			-- Transferred player's unique ID
	FROM_TEAM_ID INT,			-- Team that sold the player's ID (if applicable)
	TO_TEAM_ID INT NOT NULL,		-- Team that bought the player's ID
	CONTRACT_ID INT NOT NULL,		-- Transfer's associated contract ID
	SEASON_ID INT NOT NULL			-- Transfer's season ID
);

-- COMPETITION
CREATE TABLE COMPETITION(
	COMP_ID INT NOT NULL PRIMARY KEY,	-- Competition's unique ID
	COMP_NAME VARCHAR(50) NOT NULL,		-- Competition name
	COMP_COUNTRY VARCHAR(30) NOT NULL,	-- Country of the competition (unless continental e.g. UCL)
	COMP_TYPE ENUM('League', 'Knockout', 'Hybrid') NOT NULL,	-- Type of competition
	COMP_TIER TINYINT CHECK(COMP_TIER BETWEEN 1 AND 4),		-- Tier level of the competition
	COMP_GOVERN VARCHAR(10) NOT NULL	-- Governing body of the competition (e.g., UEFA, FA)
);

-- CONTRACT
CREATE TABLE CONTRACT(
	CONTRACT_ID INT NOT NULL PRIMARY KEY,		-- Contract unique ID
	CONTRACT_START DATE NOT NULL,			-- Contract start date
	CONTRACT_END DATE NOT NULL CHECK(CONTRACT_END > CONTRACT_START),	-- Contract end date
	CONTRACT_SALARY INT NOT NULL CHECK(CONTRACT_SALARY > 0),	-- Contract salary in dollars
	CONTRACT_BONUS INT CHECK(CONTRACT_BONUS > 0),			-- Total value in dollars of any bonuses
	CONTRACT_RELEASE INT CHECK(CONTRACT_RELEASE > 0),		-- Value in dollars of a release clause
	CONTRACT_ACTIVE BOOLEAN NOT NULL,		-- Flag if the contract is currently active
	CONTRACT_TYPE ENUM('Player', 'Manager') NOT NULL,		-- Type of contract
	PLAYER_ID INT,				-- Contract's player ID (if 'Player' contract)
	MANAGER_ID INT,				-- Contract's manager ID (if 'Manager' contract)
	TEAM_ID INT NOT NULL,			-- Contract's team ID
	TRAN_ID INT				-- Contract's transfer ID
);

-- STANDING
CREATE TABLE STANDING (
    SEASON_ID INT NOT NULL,                                -- Season ID
    COMPETITION_ID INT NOT NULL,                           -- Competition ID
    TEAM_ID INT NOT NULL,                                  -- Team ID
    STAND_GAMES SMALLINT NOT NULL CHECK (STAND_GAMES BETWEEN 0 AND 50), -- Games played
    STAND_WINS SMALLINT NOT NULL CHECK (STAND_WINS <= STAND_GAMES),     -- Wins
    STAND_DRAWS SMALLINT NOT NULL CHECK (STAND_DRAWS <= STAND_GAMES),   -- Draws
    STAND_LOSSES SMALLINT NOT NULL CHECK (STAND_LOSSES <= STAND_GAMES), -- Losses
    STAND_POINTS SMALLINT NOT NULL,                        -- Points
    STAND_GS SMALLINT NOT NULL CHECK (STAND_GS >= 0),      -- Goals scored
    STAND_GA SMALLINT NOT NULL CHECK (STAND_GA >= 0),      -- Goals against
    STAND_GD SMALLINT NOT NULL CHECK (STAND_GD BETWEEN -999 AND 999),   -- Goal differential
    STAND_POS SMALLINT NOT NULL CHECK (STAND_POS BETWEEN 1 AND 50),     -- Position
    STAND_DATE DATE NOT NULL,                              -- Last update
    PRIMARY KEY (SEASON_ID, COMPETITION_ID, TEAM_ID),      -- Composite PK
    CHECK (STAND_POINTS BETWEEN -50 AND (3 * STAND_WINS + STAND_DRAWS)), -- Valid points
    CHECK (STAND_WINS + STAND_DRAWS + STAND_LOSSES = STAND_GAMES)       -- Valid games
);

-- PLAYER_MATCH_STAT
CREATE TABLE PLAYER_MATCH_STAT(
	PLAYER_ID INT NOT NULL,		-- Associated player ID
	MATCH_ID INT NOT NULL,		-- Associated match ID
	PM_MINUTES SMALLINT NOT NULL CHECK(PM_MINUTES BETWEEN 0 AND 120),	-- # of minutes played
	PM_GOALS SMALLINT NOT NULL CHECK(PM_GOALS BETWEEN 0 AND 50),		-- # of goals scored	
	PM_ASSISTS SMALLINT NOT NULL CHECK(PM_ASSISTS BETWEEN 0 AND 50),		-- # of assists
	PM_SHOTS SMALLINT NOT NULL CHECK(PM_SHOTS BETWEEN 0 AND 99),		-- # of shots
	PM_SOG SMALLINT NOT NULL CHECK(PM_SOG BETWEEN 0 AND PM_SHOTS),		-- # of shots on goal
	PM_PKS SMALLINT NOT NULL CHECK(PM_PKS BETWEEN 0 AND PM_GOALS),		-- # of PKs scored
	PM_PKM SMALLINT NOT NULL CHECK(PM_PKM BETWEEN 0 AND PM_SHOTS),		-- # of PKs missed
	PM_PA SMALLINT NOT NULL CHECK(PM_PA BETWEEN 0 AND 400),			-- # of passes attempted
	PM_PC SMALLINT NOT NULL CHECK(PM_PC BETWEEN 0 AND PM_PA),		-- # of passes completed
	PM_TACKLES SMALLINT NOT NULL CHECK(PM_TACKLES BETWEEN 0 AND 50),	-- # of tackles
	PM_FOULS_C SMALLINT NOT NULL CHECK(PM_FOULS_C BETWEEN 0 AND 20),	-- # of fouls committed
	PM_FOULS_A SMALLINT NOT NULL CHECK(PM_FOULS_A BETWEEN 0 AND 20),	-- # of fouls against
	PM_YC SMALLINT NOT NULL CHECK(PM_YC BETWEEN 0 AND 2),			-- # of yellow cards
	PM_RC SMALLINT NOT NULL CHECK(PM_RC BETWEEN 0 AND 1),			-- # of red cards
	PM_INTERCEPT SMALLINT NOT NULL CHECK(PM_INTERCEPT BETWEEN 0 AND 50),	-- # of interceptions
	PM_SAVES SMALLINT NOT NULL CHECK(PM_SAVES BETWEEN 0 AND 50),		-- # of saves
	PM_CONCEDE SMALLINT NOT NULL CHECK(PM_CONCEDE BETWEEN 0 AND 50),	-- # of goals conceded
	PM_PK_SAVE SMALLINT NOT NULL CHECK(PM_PK_SAVE BETWEEN 0 AND PM_SAVES),	-- # of PKs saved
	PM_START BOOLEAN NOT NULL,		-- Flag if the player started
	PRIMARY KEY (PLAYER_ID, MATCH_ID)	-- Composite primary key
);

-- TEAM_MATCH_STAT
CREATE TABLE TEAM_MATCH_STAT(
	TEAM_ID INT NOT NULL,		-- Associated team ID
	MATCH_ID INT NOT NULL,		-- Associated player ID
	TM_HOME BOOLEAN NOT NULL,	-- Flag if the team is Home (1) or Away (0)
	TM_GS SMALLINT NOT NULL CHECK(TM_GS BETWEEN 0 AND 50),			-- # of goals scored
	TM_GA SMALLINT NOT NULL CHECK(TM_GA BETWEEN 0 AND 50),			-- # of goals against
	TM_POSS DECIMAL(3, 1) NOT NULL CHECK(TM_POSS BETWEEN 00.1 AND 99.9),	-- % of possession
	TM_SHOTS SMALLINT NOT NULL CHECK(TM_SHOTS BETWEEN 0 AND 99),		-- # of shots
	TM_SOG SMALLINT NOT NULL CHECK(TM_SOG BETWEEN 0 AND TM_SHOTS),		-- # of shots on goal
	TM_FOULS_C SMALLINT NOT NULL CHECK(TM_FOULS_C BETWEEN 0 AND 50),	-- # of fouls committed
	TM_FOULS_A SMALLINT NOT NULL CHECK(TM_FOULS_A BETWEEN 0 AND 50),	-- # of fouls against
	TM_YC SMALLINT NOT NULL CHECK(TM_YC BETWEEN 0 AND 40),			-- # of yellow cards
	TM_RC SMALLINT NOT NULL CHECK(TM_RC BETWEEN 0 AND 20),			-- # of red cards
	TM_PA SMALLINT NOT NULL CHECK(TM_PA BETWEEN 0 AND 1000),		-- # of passes attempted
	TM_PC SMALLINT NOT NULL CHECK(TM_PC BETWEEN 0 AND TM_PA),		-- # of passes completed
	TM_CORNER SMALLINT NOT NULL CHECK(TM_CORNER BETWEEN 0 AND 50),		-- # of corner kicks
	TM_RESULT ENUM('Win', 'Loss', 'Draw') NOT NULL,		-- Match result
	PRIMARY KEY (TEAM_ID, MATCH_ID)			-- Composite Primary Key
);

-- SEASON
CREATE TABLE SEASON(
	SEASON_ID INT NOT NULL PRIMARY KEY,		-- Season unique ID
	SEASON_START DATE NOT NULL,			-- Season start date
	SEASON_END DATE NOT NULL CHECK(SEASON_END > SEASON_START),	-- Season end date
	SEASON_ACTIVE BOOLEAN NOT NULL			-- Flag if season is active
);

-- PLAYER_TEAM_HISTORY
CREATE TABLE PLAYER_TEAM_HISTORY(
	PLAYER_ID INT NOT NULL,			-- Associated player ID
	TEAM_ID INT NOT NULL,			-- Associated team ID
	PT_START DATE NOT NULL,			-- Player/Team start date
	PT_END DATE CHECK(PT_END > PT_START),	-- Player/Team end date
	PT_JERSEY SMALLINT NOT NULL CHECK(PT_JERSEY BETWEEN 1 AND 99),	-- Player's jersey number
	PRIMARY KEY (PLAYER_ID, TEAM_ID, PT_START)	-- Composite primary key
);

-- MANAGER_TEAM_HISTORY
CREATE TABLE MANAGER_TEAM_HISTORY(
	MANAGER_ID INT NOT NULL,		-- Associated manager ID
	TEAM_ID INT NOT NULL,			-- Associated team ID
	MT_START DATE NOT NULL,			-- Manager/Team start date
	MT_END DATE CHECK(MT_END > MT_START),	-- Manager/Team end date
	PRIMARY KEY (MANAGER_ID, TEAM_ID, MT_START)	-- Composite Primary Key
);

-- Database Query Examples -- 

-- 1. What are the capacities of all English stadiums?
SELECT STADIUM_NAME, STADIUM_CITY, STADIUM_CAPACITY 
FROM STADIUM 
WHERE STADIUM_COUNTRY = 'England';

-- 2. What is the current Manchester United roster?
SELECT P.PLAYER_ID, P.PLAYER_FNAME, P.PLAYER_LNAME, P.PLAYER_ALIAS, P.PLAYER_POS, PTH.PT_JERSEY
FROM PLAYER P
JOIN PLAYER_TEAM_HISTORY PTH ON P.PLAYER_ID = PTH.PLAYER_ID
WHERE PTH.TEAM_ID =
(SELECT TEAM_ID FROM TEAM WHERE TEAM_NAME = ‘Manchester United’)
AND PTH.PT_END IS NULL;


-- 3. What is the current Premier League Table?
SELECT S.STAND_POS, T.TEAM_NAME, S.STAND_POINTS, S.STAND_GAMES,
S.STAND_WINS, S.STAND_DRAWS, S.STAND_LOSSES, S.STAND_GD,
S.STAND_GA, S.STAND_GD
FROM STANDING S
JOIN TEAM T ON S.TEAM_ID = T.TEAM_ID
JOIN COMPETITION C ON S.COMPETITION_ID = C.COMP_ID
JOIN SEASON SE ON S.SEASON_ID = SE.SEASON_ID
WHERE C.COMP_NAME = 'Premier League'
AND SE.SEASON_ACTIVE = 1
ORDER BY S.STAND_POS ASC;
